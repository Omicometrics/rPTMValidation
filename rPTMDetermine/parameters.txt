[Modification]
# >>>>> Modifications related parameters
# List of modifications excluded in the analysis, separated by `;` and no space
# between modifications. Each modification is presented as `name@site`, with
# name must be PSI-MS Name or Interim name (if PSI-MS Name does not exist) in
# Unimod database. For terminal modifications, use `nterm` for N-terminal and
# `cterm` for C-terminal modifications. For example, `Carbamidomethyl@C` will
# exclude all carbamidomethylation on cysteine, and `Acetyl@nterm` to exclude
# N-terminal acetylation. Note that, if other modifications co-exist in a
# peptide, these modificatios will also be considered during localization.
# Otherwise the identifications are excluded in the analysis.
excludes = Carbamidomethyl@C;Oxidation@M

# List of modifications considered for validation. Some of the modifications
# have been identified by database search. However, it may be possible that
# additional modifications are considered. All these modifications should be
# provided in a separate file (either `.csv` or `.txt` type file). Note that
# Excel files can't be recognized. The table is a single column with a name
# per line, no header is required. These names must be PSI-MS Name or Interim
# name (if PSI-MS Name does not exist) in Unimod database. The mass and sites
# of these modifications will be determined automatically from the Unimod
# database. Leave empty will use the modifications identified by database
# search only.
modification_list =

# Modification isobaric tolerance. The tolerance to merging modifications in
# `modification_list` or list of modifications in database search results with
# mass within the tolerance, so that if these modifications locate at the same
# sites, they will be combined and not differentiated as they will be treated
# like isobarics. As a result, they will not be reported in the final results.
# Both `Da` and `ppm` are accepted.
modification_tolerance = 0.1 Da

# Fix modifications. In Percolator post-processed results, fixed modification
# is not shown in results. This compensates for this setup, with same format
# as `excludes`.
fixed_modification = Carbamidomethyl@C
